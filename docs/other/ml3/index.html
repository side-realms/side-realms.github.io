class MyAutoEncoder(Model):
  def __init__(self, *args, **kwargs): # åŸºåº•ã‚¯ãƒ©ã‚¹ã®ã‚³ãƒ³ã‚¹ãƒˆãƒ©ã‚¯ã‚¿ã‚’ä¸Šæ›¸ãã™ã‚‹

    self.encoder = Sequential()
    self.encoder.add(layers.Flatten())
    self.encoder.add(layers.Dense(
        32,
        activation= "tanh",
        name = "encoder"))

    self.decoder = Sequential()
    self.decoder.add(layers.Dense(
        784,
        activation = "sigmoid",
        name = "decoder"))
    self.decoder.add(layers.Reshape((28, 28)))

  def call(self, inputs):
    encoded = self.encoder(inputs)
    decoder = self.decoder(encoded)
    outputs = decoder
    return outputs
