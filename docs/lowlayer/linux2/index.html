void myhead(FILE *fp, int nlines){
    int c;

    while(nlines > 0 && (c = getc(fp)) != EOF){
        if(putchar(c) < 0){
            exit(1);
        }
        if(c == '\n'){
            nlines--;
        }
        if(nlines == 0){
            return;
        }
    }
}

int main(int argc, char *argv[]){

    int nlines, opt;

    if(argc < 2){
        printf("Usage: %s <filename> <nlines>\n", argv[0]);
        exit(1);
    }

    while((opt=getopt_long(argc, argv, "n:", longopts, NULL)) != -1){
        switch(opt){
            case 'n':
                nlines = atoi(optarg);
                break;
        }
    }

    if(argc == 2){
        myhead(stdin, nlines);
        exit(0);
    } else {
        FILE *fp;
        int i;

        for(i=2; i < argc; i++){
            fp = fopen(argv[i], "r");
            if(fp == NULL){
                perror(argv[i]);
                exit(1);
            }
            myhead(fp, nlines);
        }
    }
}

int buf_counter = 0;

void mytail(int num, FILE *fp){
    char buf[num][1024];

    while(fgets(buf[buf_counter], sizeof(buf[buf_counter]), fp) != NULL){
        buf_counter = (buf_counter + 1) % num;
